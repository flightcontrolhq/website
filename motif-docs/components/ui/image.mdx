export const Image = ({ src, alt, className, rounded, center, border }) => {
  return (
    <img
      src={src}
      alt={alt}
      className={`${className} ${rounded ? 'rounded-primary' : ''} ${center ? 'mx-auto' : ''} ${
        border ? 'border border-gray-200' : ''
      }`}
    />
  )
}
