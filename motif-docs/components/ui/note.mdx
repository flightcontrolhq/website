export const getClassName = type => {
  switch (type) {
    case 'info':
      return 'bg-primary-100 border-primary-200'
    case 'warning':
      return 'bg-orange-100 border-orange-200'
    case 'error':
      return 'bg-red-100 border-red-200'
    default:
      return 'bg-gray-100 border-gray-200'
  }
}

export const Note = ({ type, children }) => {
  return <div className={`note p-3 dark:text-primary-900 rounded-primary border ${getClassName(type)}`}>{children}</div>
}

<Note>
  This is a simple note.
</Note>

<Note type="info">
  This is a simple note.
</Note>

</Note>

  This is a simple note.
</Note>
