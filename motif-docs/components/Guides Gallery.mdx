export const GalleryItem = ({ title, subtitle, cover, href }) => {
  return <div className="flex flex-col gap-1">

        <div

export const tabs = [
  "Authoring",
  "Designing",
  "Publishing"
]

export const Tab = ({ label, active, onClick }) => {
  return <div className={cn(
        "py-2 border-b-2 font-medium text-sm cursor-pointer",
        {
          "border-fuchsia-500 text-fuchsia-600": active,
          "border-transparent hover:text-fuchsia-600": !active,
        }
      )}
      onClick={onClick}
    >{ label }</div>
}
export const TabBar = ({ activeIndex, onSelect }) => {
  return <div className="flex flex-row gap-6 border-b border-gray-100">
    { tabs.map((t, i) => (
        <Tab
          label={t}
          active={activeIndex === i}
          onClick={() => onSelect(i)}
        />)
      )
    }
    </div>
}

export const loremIpsum = "Lorem ipsum dolor sit amet, consectetur adipiscing elit."

export const SampleGuidesGallery = () => {
  const [tab, setTab] = useState(0)
  return <div>
    <TabBar activeIndex={tab} onSelect={setTab} />
    <div className="grid grid-cols-1 sm:grid-cols-3 gap-6 mt-6">
      {cards.slice(tab * 3, tab * 3 + 3).map(card => {
        return <GalleryItem
        title={card.title}
        subtitle={loremIpsum}
        href="/guides"
        cover={card.cover}
      />
      })}
    </div>
  </div>
}
