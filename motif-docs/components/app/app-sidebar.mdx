import cn from "classnames"
import Logo from "@components/icons/logo"
import { getSidebarId } from "@lib/sidebar"
import config from "/project-config"
import AppSearch from "@components/app/app-search"
import { ThemeSwitcher } from "@components/ui/themeswitcher"

export const AppSidebarList = ({ activeSlug, onClick }) => {
  let sidebarId = getSidebarId(activeSlug)
  return <div className="flex flex-col flex-grow gap-8">
      {config.sidebars[sidebarId]?.map((section) => {
        const parentActive = section.href === activeSlug
        console.log('parentActive', parentActive)
        return <div className="flex flex-col" key={sidebarId}>
            {section.title && !section.href &&
              <p className="text-base sm:text-sm font-semibold px-4 mb-2">
                {section.title}
              </p>
            }
            {section.title && section.href &&
              <a
         
              href={section.href}
              onClick={onClick}
              className={cn(
                  "text-base sm:text-sm px-4 mb-2 font-semibold transition hover:text-[#090a8a]",
                )}>
                  <div className={cn(
                  "transition",
                  {
                    "text-primary-500 dark:text-primary-400 text-[#090a8a]": parentActive,
                    "text-base sm:text-sm font-semibold": !parentActive
                  }
                )}>
                    {section.title}
                  </div>
                </a>
            }
            {section.pages.map((page) => {
              const active = page.href === activeSlug
              return <a
                key={page.href}
                href={page.href}
                onClick={onClick}
                className={cn(
                    "text-base sm:text-sm px-4 transition",
                  )}>
                    <div className={cn(
                    "border-l px-3 py-1.5 transition ",
                    {
                      "dark:text-primary-400 border-primary-500 text-[#090a8a]": active,
                      "text-neutral-700 hover:text-[#090a8a] dark:text-white/70 hover:dark:text-white/90 border-neutral-200 dark:border-white/20 hover:border-[#090a8a] dark:hover:border-white/50": !active
                    }
                  )}>
                      {page.title}
                    </div>
                  </a>
            })}
          </div>
      })}
    </div>
}

export const AppSidebar = ({ activeSlug, searchData, noHeader, noSearch }) => {
  return (
    <div className="z-10 hidden md:block fixed inset-0 w-0 md:w-64 pt-24">
      <div className="absolute w-full">
        <div className="h-6 bg-gradient-to-b from-white to-white/0 dark:from-neutral-900 dark:to-neutral-900/0"/>
      </div>
      <div className="overflow-y-auto h-full pb-10 pt-[49px] hiddenScrollbar pl-4">
        <AppSidebarList activeSlug={activeSlug} />
      </div>
    </div>
  )
}
